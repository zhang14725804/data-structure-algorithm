### 模板1

```bash
func bsearch1(l,r int){
    while(l<r){
        mid := (l+r) >> 1
        if check(mid){
            r = mid
        }else{
            l = mid+1
        }
    }
    return l
}
```

### 模板2

```bash
func bsearch2(l,r int){
    while(l<r){
        mid := (l+r+1) >> 1
        if check(mid){
            l = mid
        }else{
            r = mid-1
        }
    }
    return l
}
```
